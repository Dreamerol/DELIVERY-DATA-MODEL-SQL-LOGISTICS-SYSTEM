CREATE TABLE clients(
    
    NAME VARCHAR(32) NOT NULL,
    EGN VARCHAR(32)NOT NULL,
    ORDER_NUMBER INT NOT NULL,
    BANK_ACCOUNT VARCHAR(32) NOT NULL UNIQUE,
    DELIVER_NUMBER INT NOT NULL UNIQUE,
    PRIMARY KEY(EGN)
);

USE Dostavki;
CREATE TABLE bank_accounts(
    
    SUM_OF_MONEY DECIMAL,
    TYPE_ACCOUNT VARCHAR(32) NOT NULL,
    BANK_ACCOUNT_NUMBER VARCHAR(32) UNIQUE NOT NULL,
    PRIMARY KEY(BANK_ACCOUNT_NUMBER)
);
